/**
 An object within a {{#crossLink "GLTFModel"}}{{/crossLink}}.

 * A convenience component to make scripting {{#crossLink "GLTFModel"}}{{/crossLink}}s easier.
 * Corresponds to a glTF node in mesh.
 * Has an {{#crossLink "Entity"}}{{/crossLink}} for each primitive in that glTF mesh.
 * Has convenience properties through which you can set and get things on entities collectively, as if they were one object,

 @class GLTFObject
 @module xeogl
 @submodule models
 @constructor
 @extends Component
 */
xeogl.GLTFObject = xeogl.Component.extend({

    /**
     JavaScript class name for this xeogl.GLTFObject.

     @property type
     @type String
     @final
     */
    type: "xeogl.GLTFObject",

    // Constructor

    _init: function (cfg) {
        this.model = cfg.model;
        this.ifcType = cfg.ifcType || "DEFAULT";

        this._aabb = null; // Lazy-create
        this._aabbDirty = false;
        this._entities = [];

        if (cfg.entities) {
            var entities = cfg.entities;
            if (xeogl._isArray(entities)) {
                for (var i = 0, len = entities.length; i < len; i++) {
                    this._addEntity(entities[i]);
                }
            } else {
                for (var id in entities) {
                    if (entities.hasOwnProperty(id)) {
                        this._addEntity(entities[id]);
                    }
                }
            }
        }
    },

    /**
     * Adds a primitive entity to this GLTFObject
     * @param {xeogl.Entity} mesh
     */
    _addEntity: function (entity) {
        this._entities.push(entity);
        this._aabbDirty = true;
    },

    _update: function () {

        // Update the AABB if any entities were added or moved
        // Lazy-create AABB on demand to save memory

        if (!this._aabb) {
            this._aabb = xeogl.math.AABB3();
            this._aabbDirty = true;
            var entity;
            for (var i = 0, len = this._entities.length; i < len; i++) {
                entity = this._entities[i];
                entity.worldBoundary.on("updated", function () {
                    this._aabbDirty = true;
                }, this);
            }
        }
        if (this._aabbDirty) {
            xeogl.math.collapseAABB3(this._aabb);
            for (var i = 0, len = this._entities.length; i < len; i++) {
                xeogl.math.expandAABB3(this._aabb, this._entities[i].worldBoundary.aabb);
            }
            if (!this._center) {
                this._center = new Float32Array(3);
            }
            xeogl.math.getAABB3Center(this._aabb, this._center);
            this._aabbDirty = false;
        }
    },

    _buildTransform: (function () {

        // Lazy-rebuilds the transform hierarchy shared by the entities

        var objectCenter = new Float32Array(3);
        var matrix = new Float32Array(16);
        var offset = new Float32Array(3);
        var negOffset = new Float32Array(3);

        return function (object) {

            if (this._scale) {
                return;
            }

            if (this._entities.length === 0) {
                return;
            }

            var scene = this.scene;
            var math = xeogl.math;

            math.getAABB2Center(this.aabb, objectCenter);
            var sceneCenter = scene.worldBoundary.center;
            math.subVec3(objectCenter, sceneCenter, offset);
            math.mulVec3Scalar(offset, -1, negOffset);

            var entity0 = this._entities[0];

            var modelTransform = model.transform;
            math.identityMat4(matrix);
            for (var t = entity0.transform; t.id !== modelTransform.id; t = t.parent) {
                math.mulMat4(matrix, t.matrix, matrix);
            }

            var transform = this._scale = new xeogl.Scale(this, {
                parent: new xeogl.Transform(this, {
                    matrix: matrix,
                    parent: new xeogl.Translate(this, {
                        xyz: negOffset,
                        parent: this._rotate = new xeogl.Quaternion(this, {
                            parent: this._translate = new xeogl.Translate(this, {
                                parent: new xeogl.Translate(this, {
                                    xyz: offset,
                                    parent: model.transform
                                })
                            })
                        })
                    })
                })
            });

            // Share the transform among the entities
            for (var i = 0, len = this._entities.length; i < len; i++) {
                this._entities[i].transform = transform;
            }
        };
    })(),

    _props: {

        /**
         * The axis-aligned World-space boundary of this GLTFObject.
         * Dynamically encloses all the entities.
         */
        aabb: {
            get: function () {
                this._update();
                return this._aabb;
            }
        },

        /**
         * The World-space center of this GLTFObject.
         * @returns {Float32Array}
         */
        center: function () {
            this._update();
            return this._center;
        },

        /**
         * Whether this GLTFObject is visible or not
         */
        visible: {
            set: function (visible) {
                for (var i = 0, len = this._entities.length; i < len; i++) {
                    this._entities[i].visible = visible;
                }
            },

            get: function () {
                return this._entities[0].visible;
            }
        },

        /**
         * The opacity of this GLTFObject
         */
        opacity: {
            set: function (opacity) {
                var material;
                for (var i = 0, len = this._entities.length; i < len; i++) {
                    material = this._entities[i].material;
                    material.alpha = opacity;
                    material.alphaMode = (opacity < 1.0) ? "blend" : "opaque";
                }
            },

            get: function () {
                return this._entities[0].material.alpha;
            }
        },

        /**
         * Whether or not this GLTFObject is clippable
         */
        clippable: {
            set: function (clippable) {
                for (var i = 0, len = this._entities.length; i < len; i++) {
                    this._entities[i].clippable = clippable;
                }
            },

            get: function () {
                return this._entities[0].clippable;
            }
        },

        /**
         * Whether or not this GLTFObject is pickable
         */
        pickable: {
            set: function (pickable) {
                for (var i = 0, len = this._entities.length; i < len; i++) {
                    this._entities[i].pickable = pickable;
                }
            },

            get: function () {
                return this._entities[0].pickable;
            }
        },

        /**
         * Whether or not this GLTFObject is outlined
         */
        outlined: {
            set: function (outlined) {
                for (var i = 0, len = this._entities.length; i < len; i++) {
                    this._entities[i].outlined = outlined;
                }
            },

            get: function () {
                return this._entities[0].outlined;
            }
        },

        /**
         * The entities in this GLTFObject
         */
        entities: {
            get: function () {
                return this._entities;
            }
        },

        ///**
        // * The modelling transform
        // */
        //transform: {
        //    set: function (transform) {
        //        for (var i = 0, len = this._entities.length; i < len; i++) {
        //            this._entities[i].transform = transform; // Share the transform among the entities
        //        }
        //    },
        //
        //    get: function () {
        //        return this._entities[0].transform;
        //    }
        //},

        translate: {
            set: function (xyz) {
                this._buildTransform();
                this._translate.xyz = xyz;
            },

            get: function () {
                this._buildTransform();
                return this._translate.xyz;
            }
        },

        rotate: {
            set: function (xyz) {
                this._buildTransform();
            },

            get: function () {
                this._buildTransform();
            }
        },

        scale: {
            set: function (xyz) {
                this._buildTransform();
                this._scale.xyz = xyz;
            },

            get: function () {
                this._buildTransform();
                return this._scale.xyz;
            }
        }
    },

    /**
     * Increments or decrements the translation of this GLTFObject.
     *
     * @param {[Number, Number, Number]} xyz World-space translation vector.
     * @returns {BIMObject} this
     */
    addTranslate: function (xyz) {
        var xyzOld = this.translate;
        this.translate = [xyzOld[0] + xyz[0], xyzOld[1] + xyz[1], xyzOld[2] + xyz[2]];
        return this;
    }

});
