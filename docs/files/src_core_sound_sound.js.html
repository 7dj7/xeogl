/**
 A **Sound** is a directional light source that illuminates all attached {{#crossLink "GameObject"}}GameObjects{{/crossLink}} equally
 from a given direction.

 ## Overview

 <ul>
 <li>DirLights are grouped, along with other light source types, within {{#crossLink "Lights"}}Lights{{/crossLink}} components,
 which are attached to {{#crossLink "GameObject"}}GameObjects{{/crossLink}}.</li>
 <li>DirLights have a direction, but no position.</li>
 <li>DirLights may be defined in either **World** or **View** coordinate space. When in World-space, their direction
 is relative to the World coordinate system, and will appear to move as the {{#crossLink "Camera"}}{{/crossLink}} moves.
 When in View-space, their direction is relative to the View coordinate system, and will behave as if fixed to the viewer's
 head as the {{#crossLink "Camera"}}{{/crossLink}} moves.</li>
 <li>See <a href="Shader.html#inputs">Shader Inputs</a> for the variables that DirLights create within xeoEngine's shaders.</li>
 </ul>

 <img src="../../../assets/images/Sound.png"></img>

 ## Example

 In this example we have:

 <ul>
 <li>a {{#crossLink "PhongMaterial"}}{{/crossLink}},</li>
 <li>a Sound that points along the negative diagonal of the View coordinate system,</li>
 <li>a {{#crossLink "Lights"}}{{/crossLink}} containing the Sound,</li>
 <li>a {{#crossLink "Geometry"}}{{/crossLink}} that is the default box shape, and
 <li>a {{#crossLink "GameObject"}}{{/crossLink}} attached to all of the above.</li>
 </ul>

 <iframe style="width: 600px; height: 400px" src="../../examples/light_DirLight.html"></iframe>

 ```` javascript
 var scene = new XEO.Scene();

 // A shiny PhongMaterial with quantities of reflected
 // ambient, diffuse and specular color
 var material = new XEO.PhongMaterial(scene, {
    ambient:    [0.3, 0.3, 0.3],
    diffuse:    [0.7, 0.7, 0.7],
    specular:   [1. 1, 1],
    shininess:  30
});

 // Sound with color and intensity, pointing along
 // the negative diagonal within the View coordinate system
 var dirLight = new XEO.Sound(scene, {
    volume:         [-1, -1, -1],
    color:       [0.5, 0.7, 0.5],
    intensity:   1.0,
    space:      "view"  // Other option is "world", for World-space
});

 // Lights which contains our Sound
 var lights = new XEO.Lights(scene, {
    lights: [
        dirLight
    ]
});

 var geometry = new XEO.Geometry(scene);  // Defaults to a 2x2x2 box

 // Object which renders our Geometry, colored with
 // the Material and illuminated with the Sound
 var object = new XEO.GameObject(scene, {
    lights: lights,
    material: material,
    geometry: geometry
});
 ````

 As with all components, we can observe and change properties on a DirLights, like so:

 ````Javascript
 // Attach a change listener to a property
 var handle = dirLight.on("color",
 function(value) {
        // Property value has changed
    });

 // Set the property, which fires our change listener
 dirLight.color = [0.0, 0.3, 0.3];

 // Detach the change listener
 dirLight.off(handle);
 ````

 @class Sound
 @module XEO
 @submodule lighting
 @constructor
 @param [scene] {Scene} Parent {{#crossLink "Scene"}}Scene{{/crossLink}}, creates this Sound within the
 default {{#crossLink "Scene"}}Scene{{/crossLink}} when omitted
 @param [cfg] {*} The Sound configuration
 @param [cfg.id] {String} Optional ID, unique among all components in the parent {{#crossLink "Scene"}}Scene{{/crossLink}}, generated automatically when omitted.
 @param [cfg.meta] {String:Object} Optional map of user-defined metadata to attach to this Sound.
 @param [cfg.volume=[1.0, 1.0, 1.0]] {Array(Number)} A unit vector indicating the direction of illumination, given in either World or View space, depending on the value of the **space** parameter.
 @param [cfg.color=[0.7, 0.7, 0.8 ]] {Array(Number)} The color of this Sound.
 @param [cfg.intensity=1.0 ] {Number} The intensity of this Sound.
 @param [cfg.space="view"] {String} The coordinate system the Sound is defined in - "view" or "space".

 @extends Component
 */
(function () {

    "use strict";

    XEO.Sound = XEO.Component.extend({

        type: "XEO.Sound",

        _init: function (cfg) {

            this._playing = false;

            this.src = cfg.src;
            this.volume = cfg.volume;
            this.loop = cfg.loop;
            this.refDistance = cfg.refDistance;
            this.rolloffFactor = cfg.rolloffFactor;
            this.space = cfg.space;
        },

        _props: {

            /**
             The src of this Sound.

             Fires a {{#crossLink "Sound/src:event"}}{{/crossLink}} event on change.

             @property src
             @default null
             @type String
             */
            src: {

                set: function (value) {

                    this._src = value;

                    var scope = this;

                    var request = new XMLHttpRequest();
                    request.open( 'GET', file, true );
                    request.responseType = 'arraybuffer';
                    request.onload = function ( e ) {

                        scope.context.decodeAudioData( this.response, function ( buffer ) {

                            scope.source.buffer = buffer;

                            if( scope.autoplay ) scope.play();

                        } );

                    };
                    request.send();


                    /**
                     * Fired whenever this Sound's  {{#crossLink "Sound/src:property"}}{{/crossLink}} property changes.
                     * @event src
                     * @param value The property's new value
                     */
                    this.fire("src", this._src);
                },

                get: function () {
                    return this._src;
                }
            },
            
            /**
             The volume of this Sound.

             Fires a {{#crossLink "Sound/volume:event"}}{{/crossLink}} event on change.

             @property volume
             @default 1.0
             @type Number
             */
            volume: {

                set: function (value) {

                    value = value !== undefined ? value : 1.0;

                    this._volume = value;

                    this._gain.gain.value = value;

                    /**
                     * Fired whenever this Sound's  {{#crossLink "Sound/volume:property"}}{{/crossLink}} property changes.
                     * @event volume
                     * @param value The property's new value
                     */
                    this.fire("volume", this._volume);
                },

                get: function () {
                    return this._volume;
                }
            },

            /**
             Indicates whether this sound loops.

             Fires a {{#crossLink "Sound/loop:event"}}{{/crossLink}} event on change.

             @property loop
             @default true
             @type Boolean
             */
            loop: {

                set: function (value) {

                    value = value !== undefined ? value : true;

                    this._loop = value;

                    this._source.loop = value;

                    /**
                     * Fired whenever this Sound's  {{#crossLink "Sound/loop:property"}}{{/crossLink}} property changes.
                     * @event loop
                     * @param value The property's new value
                     */
                    this.fire("loop", this._loop);
                },

                get: function () {
                    return this._loop;
                }
            },

            /**
             The roll-off factor of this Sound.

             Fires a {{#crossLink "Sound/rolloffFactor:event"}}{{/crossLink}} event on change.

             @property rolloffFactor
             @default 1.0
             @type Number
             */
            rolloffFactor: {

                set: function (value) {

                    value = value !== undefined ? value : 1.0;

                    this._rolloffFactor = value;

                    this._gain.gain.value = value;

                    /**
                     * Fired whenever this Sound's  {{#crossLink "Sound/rolloffFactor:property"}}{{/crossLink}} property changes.
                     * @event rolloffFactor
                     * @param value The property's new value
                     */
                    this.fire("rolloffFactor", this._rolloffFactor);
                },

                get: function () {
                    return this._rolloffFactor;
                }
            },

            /**
             The reference distance of this Sound.

             Fires a {{#crossLink "Sound/refDistance:event"}}{{/crossLink}} event on change.

             @property refDistance
             @default 1.0
             @type Number
             */
            refDistance: {

                set: function (value) {

                    value = value !== undefined ? value : 1.0;

                    this._refDistance = value;

                    this._gain.gain.value = value;

                    /**
                     * Fired whenever this Sound's  {{#crossLink "Sound/refDistance:property"}}{{/crossLink}} property changes.
                     * @event refDistance
                     * @param value The property's new value
                     */
                    this.fire("refDistance", this._refDistance);
                },

                get: function () {
                    return this._refDistance;
                }
            },

            /**
             Indicates if this sound is playing.

             Fires a {{#crossLink "Sound/playing:event"}}{{/crossLink}} event on change.

             @property playing
             @default 1.0
             @type Number
             @final
             */
            playing: {

                get: function () {
                    return this._playing;
                }
            }
        },

        play: function () {

            if (this._playing === true) {
                this.warn("Already playing");
                return;

            }

            var source = this._context.createBufferSource();

            source.buffer = this._source.buffer;
            source.loop = this._source.loop;
            source.onended = this._source.onended;
            source.connect(this._panner);
            source.start(0, this._startTime);

            this._playing = true;

            this._source = source;
        },

        pause: function () {
            this._source.stop();
            this._startTime = this.context.currentTime;
        },

        stop: function () {
            this._source.stop();
            this._startTime = 0;
        },

        _getJSON: function () {
            return {
                src: this._src,
                refDistance: this._refDistance,
                rolloffFactor: this._rolloffFactor,
                volume: this._volume,
                loop: this._loop
            };
        }
    });

})();
