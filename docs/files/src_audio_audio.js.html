/**

 @class Audio
 @module XEO
 @submodule audio
 @constructor
 @param [scene] {Scene} Parent {{#crossLink "Scene"}}Scene{{/crossLink}}, creates this Audio within the
 default {{#crossLink "Scene"}}Scene{{/crossLink}} when omitted
 @param [cfg] {*} The Audio configuration
 @param [cfg.id] {String} Optional ID, unique among all components in the parent {{#crossLink "Scene"}}Scene{{/crossLink}}, generated automatically when omitted.
 @param [cfg.meta] {String:Object} Optional map of user-defined metadata to attach to this Audio.
 @param [cfg.volume=[1.0, 1.0, 1.0]] {Array(Number)} A unit vector indicating the direction of illumination, given in either World or View space, depending on the value of the **space** parameter.
 @param [cfg.color=[0.7, 0.7, 0.8 ]] {Array(Number)} The color of this Audio.
 @param [cfg.intensity=1.0 ] {Number} The intensity of this Audio.
 @param [cfg.space="view"] {String} The coordinate system the Audio is defined in - "view" or "space".

 @extends Component
 */
(function () {

    "use strict";

    XEO.Audio = XEO.Component.extend({

        type: "XEO.Audio",

        _init: function (cfg) {

            this._listenerDirty = false;
            this._srcDirty = true;
            this._posDirty = true;
            this._volumeDirty = true;
            this._loopDirty = true;

            this._playing = false;

            this.listener = cfg.listener;
            this.autoplay = cfg.autoplay;
            this.src = cfg.src;
            this.pos = cfg.pos;
            this.volume = cfg.volume;
            this.loop = cfg.loop;
            this.rolloffFactor = cfg.rolloffFactor;
            this.refDistance = cfg.refDistance;
        },

        _setAudioDirty: function () {
            if (!this._audioDirty) {
                this._audioDirty = true;
                var self = this;
                this.scene.once("tick",
                    function () {
                        self._buildAudio();
                    });
            }
        },

        _buildAudio: function () {

            this._audioDirty = false;

            var self = this;

            if (this._listenerDirty) {

                this._context = this._listener.context;
                this._source = this._context.createBufferSource();
                //this._source.onended = this.onEnded.bind(this);

                this._gain = this._context.createGain();
                this._gain.connect( this._context.destination );

                this._panner = this._context.createPanner();
                this._panner.connect( this._gain );

                this._listenerDirty = false;
                this._srcDirty = true;
                this._posDirty = true;
                this._volumeDirty = true;
                this._loopDirty = true;
            }

            if (this._srcDirty) {
                var request = new XMLHttpRequest();
                request.open('GET', file, true);
                request.responseType = 'arraybuffer';
                request.onload = function (e) {

                    self._context.decodeAudioData(this.response,
                        function (buffer) {

                            self._source.buffer = buffer;

                            if (self._autoplay) {
                                self.play();
                            }

                            self._srcDirty = false;
                        });
                };
                request.send();
                this._srcDirty = false;
            }

            if (this._posDirty) {
                this._context.listener.set(this._pos[0], this._pos[1], this._pos[2])
                this._posDirty = false;
            }

            if (this._volumeDirty) {
                this._gain.gain.value = value;
                this._volumeDirty = false;
            }

            if (this._loopDirty) {
                this._source.loop = value;
                this._loopDirty = false;
            }
        },


        _props: {

            /**
             * The Listener component for this Audio.
             *
             * When set to a null or undefined value, will default to the parent {{#crossLink "Scene"}}Scene{{/crossLink}}'s
             * default {{#crossLink "Scene/listener:property"}}listener{{/crossLink}}, which is
             * a {{#crossLink "Perspective"}}Perspective{{/crossLink}}.
             *
             * Fires a {{#crossLink "Audio/listener:event"}}{{/crossLink}} event on change.
             *
             * @property listener
             * @type Listener
             */
            listener: {

                set: function (value) {

                    this._listenerDirty = true;

                    this._setAudioDirty();

                    /**
                     * Fired whenever this Audio's {{#crossLink "Audio/listener:property"}}{{/crossLink}} property changes.
                     * @event listener
                     * @param value The property's new value
                     */
                    this._setChild("listener", value);
                },

                get: function () {
                    return this._children.listener;
                }
            },

            /**
             Indicates whether this Audio automatically plays when loaded.

             Fires a {{#crossLink "Audio/autoplay:event"}}{{/crossLink}} event on change.

             @property autoplay
             @default false
             @type Boolean
             */
            autoplay: {

                set: function (value) {

                    this._autoplay = !!value;

                    /**
                     * Fired whenever this Audio's  {{#crossLink "Audio/autoplay:property"}}{{/crossLink}} property changes.
                     * @event autoplay
                     * @param value The property's new value
                     */
                    this.fire("autoplay", this._autoplay);
                },

                get: function () {
                    return this._autoplay;
                }
            },

            /**
             The src of this Audio.

             Fires a {{#crossLink "Audio/src:event"}}{{/crossLink}} event on change.

             @property src
             @default null
             @type String
             */
            src: {

                set: function (value) {

                    this._src = value;

                    this._srcDirty = true;

                    this._setAudioDirty();

                    /**
                     * Fired whenever this Audio's  {{#crossLink "Audio/src:property"}}{{/crossLink}} property changes.
                     * @event src
                     * @param value The property's new value
                     */
                    this.fire("src", this._src);
                },

                get: function () {
                    return this._src;
                }
            },

            /**
             * World-space position of this Audio.
             * .
             * Fires an {{#crossLink "Audio/pos:event"}}{{/crossLink}} event on change.
             *
             * @property pos
             * @default [0,0,0]
             * @type {Array of Number}
             */
            pos: {

                set: function (value) {

                    this._pos = value || [1, 1, 1];

                    this._posDirty = true;

                    this._setAudioDirty();

                    /**
                     Fired whenever this Translate's {{#crossLink "Translate/pos:property"}}{{/crossLink}} property changes.
                     @event pos
                     @param value {Array of Number} The property's new value
                     */
                    this.fire("pos", this._pos);
                },

                get: function () {
                    return this._pos;
                }
            },

            /**
             The volume of this Audio.

             Fires a {{#crossLink "Audio/volume:event"}}{{/crossLink}} event on change.

             @property volume
             @default 1.0
             @type Number
             */
            volume: {

                set: function (value) {

                    value = value !== undefined ? value : 1.0;

                    this._volume = value;

                    this._volumeDirty = true;

                    /**
                     * Fired whenever this Audio's  {{#crossLink "Audio/volume:property"}}{{/crossLink}} property changes.
                     * @event volume
                     * @param value The property's new value
                     */
                    this.fire("volume", this._volume);
                },

                get: function () {
                    return this._volume;
                }
            },

            /**
             Indicates whether this sound loops.

             Fires a {{#crossLink "Audio/loop:event"}}{{/crossLink}} event on change.

             @property loop
             @default true
             @type Boolean
             */
            loop: {

                set: function (value) {

                    value = value !== undefined ? value : true;

                    this._loop = value;

                    this._loopDirty = true;

                    /**
                     * Fired whenever this Audio's  {{#crossLink "Audio/loop:property"}}{{/crossLink}} property changes.
                     * @event loop
                     * @param value The property's new value
                     */
                    this.fire("loop", this._loop);
                },

                get: function () {
                    return this._loop;
                }
            },

            /**
             The roll-off factor of this Audio.

             Fires a {{#crossLink "Audio/rolloffFactor:event"}}{{/crossLink}} event on change.

             @property rolloffFactor
             @default 1.0
             @type Number
             */
            rolloffFactor: {

                set: function (value) {

                    value = value !== undefined ? value : 1.0;

                    this._rolloffFactor = value;

                    this._gain.gain.value = value;

                    /**
                     * Fired whenever this Audio's  {{#crossLink "Audio/rolloffFactor:property"}}{{/crossLink}} property changes.
                     * @event rolloffFactor
                     * @param value The property's new value
                     */
                    this.fire("rolloffFactor", this._rolloffFactor);
                },

                get: function () {
                    return this._rolloffFactor;
                }
            },

            /**
             The reference distance of this Audio.

             Fires a {{#crossLink "Audio/refDistance:event"}}{{/crossLink}} event on change.

             @property refDistance
             @default 1.0
             @type Number
             */
            refDistance: {

                set: function (value) {

                    value = value !== undefined ? value : 1.0;

                    this._refDistance = value;

                    this._gain.gain.value = value;

                    /**
                     * Fired whenever this Audio's  {{#crossLink "Audio/refDistance:property"}}{{/crossLink}} property changes.
                     * @event refDistance
                     * @param value The property's new value
                     */
                    this.fire("refDistance", this._refDistance);
                },

                get: function () {
                    return this._refDistance;
                }
            },

            /**
             Indicates if this sound is playing.

             @property playing
             @type Boolean
             @final
             */
            playing: {

                get: function () {
                    return this._playing;
                }
            }
        },

        play: function () {

            if (this._playing === true) {
                this.warn("Already playing");
                return;
            }

            var source = this._context.createBufferSource();

            source.buffer = this._source.buffer;
            source.loop = this._source.loop;
            source.onended = this._source.onended;
            source.connect(this._panner);
            source.start(0, this._startTime);

            this._playing = true;

            this._source = source;
        },

        pause: function () {

            if (this._playing !== true || !this._source) {
                return;
            }

            this._source.stop();
            this._startTime = this.context.currentTime;
        },

        stop: function () {

            if (this._playing !== true || !this._source) {
                return;
            }

            this._source.stop();
            this._startTime = 0;
        },

        _getJSON: function () {
            return {
                autoplay: this._autoplay,
                src: this._src,
                pos: this.pos,
                volume: this._volume,
                loop: this._loop,
                rolloffFactor: this._rolloffFactor,
                refDistance: this._refDistance
            };
        }
    });

})();
